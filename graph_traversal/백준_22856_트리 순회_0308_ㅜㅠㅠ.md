### 코드
```python
import sys
input=sys.stdin.readline

# 재귀 깊이 늘리기
sys.setrecursionlimit(10**9)

# 먼저, 함수
def DFS1(Node):
    global count
    visit[Node]=True

    if not visit[Tree[Node][0]] and Tree[Node][0]!=-1:
        DFS1(Tree[Node][0])
        count += 1
    if not visit[Tree[Node][1]] and Tree[Node][1]!=-1:
        DFS1(Tree[Node][1])
        count += 1

def DFS2(Node):
    global count2
    visit[Node]=True

    if not visit[Tree[Node][1]] and Tree[Node][1]!=-1:
        DFS2(Tree[Node][1])
        count2 += 1
        
# 이제 해보자
N=int(input())
Tree={}

for i in range(N):
    a,b,c=map(int,input().split())
    Tree[a]=[b,c]

visit=[False]*(N+1)
count=0
DFS1(1)

visit=[False]*(N+1)
count2=0
DFS2(1)

print(2*count-count2)
```
